159  def canny(img):
160      # Blur and grey out the image, and crop out the ROI
161      gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
162      blur_gray = cv2.medianBlur(gray, 5)
163      edges_roi = cv2.bitwise_and(blur_gray, blur_gray, mask=roi_mask)
164  
165      # Put on Canny filter
166      edges = cv2.Canny(image=edges_roi,
167                        threshold1=120,
168                        threshold2=150,
169                        apertureSize=3)
170  
171      # Do a Houghlines algorithm
172      lines = cv2.HoughLinesP(
173          edges,  # Input edge image
174          5,  # Distance resolution in pixels
175          np.pi / 180,  # Angle resolution in radians
176          threshold=100,  # Min number of votes for valid line
177          minLineLength=10,  # Min allowed length of line
178          maxLineGap=50  # Max allowed gap between line for joining them
179      )
180  
181      # Variables for combining the lines given
182      line_image = np.copy(img) * 0
183      lines_list = []
184  
185      slope_ave_r = 0
186      amount_of_lines_r = 0
187      intercept_ave_r = 0
188      intercept_ave_r_left = 0
189      tot_length_r = 0
190  
191      slope_ave_l = 0
192      amount_of_lines_l = 0
193      intercept_ave_l = 0
194      tot_length_l = 0
195  
196      i = 0
197  
198      # Iterate over points
199      for points in lines:
200          x1, y1, x2, y2 = points[0]
201          if x2 < x1:
202              temp_x1 = x1
203              temp_y1 = y1
204              x1 = x2
205              y1 = y2
206              x2 = temp_x1
207              y2 = temp_y1
208  
209          if x1 != x2:
210              slope = (y1 - y2) / (x1 - x2)
211              intercept_y_axis = y1 - (x1 * slope)
212  
213          else:
214              slope = 0
215              intercept_y_axis = 0
216  
217          # and x1 <= img.shape[1] / 2 and x2 <= img.shape[1] / 2
218  
219          if slope < 0:
220  
221              length_l = math.sqrt(((y1 - y2) ** 2) + ((x1 - x2) ** 2))
222              tot_length_l += length_l
223              slope_ave_l += slope * length_l
224              intercept_ave_l += intercept_y_axis * length_l
225  
226              amount_of_lines_l += 1
227  
228              cv2.line(line_image, (x1, y1), (x2, y2), (255, 0, 0), 4)
229  
230          # and x1 >= img.shape[1] / 2 and x2 >= img.shape[1] / 2
231  
232          elif slope >= 0:
233  
234              length_r = math.sqrt(((y1 - y2) ** 2) + ((x1 - x2) ** 2))
235              tot_length_r += length_r
236              slope_ave_r += slope * length_r
237              intercept_ave_r_left += intercept_y_axis * length_r
238              intercept_ave_r += (framex*slope + intercept_y_axis) * length_r
239  
240              amount_of_lines_r += 1
241  
242              cv2.line(line_image, (x1, y1), (x2, y2), (0, 0, 255), 4)
243  
244          else:
245              cv2.line(line_image, (x1, y1), (x2, y2), (0, 255, 0), 4)
246  
247          i += 1
248          if i == len(lines) and amount_of_lines_l > 0:
249              slope_ave_l = (slope_ave_l / amount_of_lines_l) / (tot_length_l / amount_of_lines_l)
250              slope_ave_r = (slope_ave_r / amount_of_lines_r) / (tot_length_r / amount_of_lines_r)
251  
252              intercept_ave_l = int((intercept_ave_l / amount_of_lines_l) / (tot_length_l / amount_of_lines_l))
253              intercept_ave_r = int((intercept_ave_r / amount_of_lines_r) / (tot_length_r / amount_of_lines_r))
254              intercept_ave_r_left = int((intercept_ave_r_left / amount_of_lines_r) / (tot_length_r / amount_of_lines_r))
255  
256              line_l_x2 = int((roi_pts[3][1] - intercept_ave_l)/slope_ave_l)
257              color_left = (249, 171, 59)
258              cv2.line(line_image, (0, intercept_ave_l), (line_l_x2, int(roi_pts[3][1])), color_left, 8)
259  
260              line_r_x2 = abs(int((roi_pts[3][1] - intercept_ave_r_left) / slope_ave_r))
261              color_right = (126, 138, 249)
262              cv2.line(line_image, (int(framex), intercept_ave_r), (line_r_x2, int(roi_pts[3][1])), color_right, 8)
263  
264              intercept_middle = int((intercept_ave_r + intercept_ave_l)/2)
265              line_middle_x2 = int((line_r_x2 + line_l_x2)/2)
266              color_middle = (92, 247, 133)
267              cv2.line(line_image, (int(framex*0.5), intercept_middle), (line_middle_x2, int(roi_pts[3][1])), color_middle, 20)
268              middle_line_angle = -(math.degrees(math.atan((intercept_middle - int(roi_pts[3][1]))/(int(framex*0.5) - line_middle_x2))))
269              cv2.putText(line_image, text=f'{middle_line_angle}',
270                          org=(int(framex*0.5), intercept_middle),
271                          fontFace=cv2.FONT_HERSHEY_SIMPLEX, fontScale=2, color=color_middle,
272                          thickness=5)
273              print("real angle: " + str(middle_line_angle.__round__(4)))
274              try:
275                  given_angle = cv2.getTrackbarPos('angle', 'window')
276              except:
277                  given_angle = middle_line_angle
278              print(f"given angle: {given_angle.__round__(4)}")
279  
280              middle_line_angle_corrected = (((given_angle * 2)/180) - 1).__round__(4)
281              cv2.arrowedLine(line_image, arrow_pts, [arrow_pts[0] + int(300 * middle_line_angle_corrected), arrow_pts[1]],
282                              (0, 209, 250), thickness=8)
283              cv2.putText(line_image, text=f'{middle_line_angle_corrected}',
284                          org=arrow_pts,
285                          fontFace=cv2.FONT_HERSHEY_SIMPLEX, fontScale=2, color=(0, 209, 250),
286                          thickness=5)
287  
288              if middle_line_angle > 100:
289                  vgp.left_joystick_float(x_value_float=middle_line_angle_corrected, y_value_float=0)
290                  vgp.update()
291                  print(f"steer right: {middle_line_angle_corrected}")
292              elif middle_line_angle < 80:
293                  vgp.left_joystick_float(x_value_float=middle_line_angle_corrected, y_value_float=0)
294                  vgp.update()
295                  print(f"steer left: {middle_line_angle_corrected}")
296              else:
297                  print(f"dont steer: {middle_line_angle_corrected}")
298  
299          # Maintain a simples lookup list for points
300          lines_list.append([(x1, y1), (x2, y2)])
301  
302      cv2.polylines(line_image, [mask_pts], True, (0, 137, 255), 8)
303      cv2.polylines(line_image, [roi_pts], True, (255, 150, 255), 4)
304      cv2.circle(line_image, arrow_pts, 8, (0, 255, 255), lineType=cv2.FILLED, thickness=-8)
305      lines_edges = cv2.addWeighted(frame, 1, line_image, 0.7, 0)
306  
307      return lines_edges

310  def keyout_mask_function(img):
311      cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
312      hsv_state = 0  # h = 0, s = 1, v = 2
313      h = []
314      s = []
315      v = []
316      # h_lowest = 360
317      # h_highest = 0
318  
319      img = img[mask_pts[0][1]:mask_pts[2][1], mask_pts[0][0]:mask_pts[2][0]]
320  
321      for pixel in np.nditer(img):
322          if hsv_state == 0:
323              h.append(pixel)
324  
325              # if pixel <= h_lowest:
326              #     h_lowest = pixel
327              # if pixel >= h_highest:
328              #     h_highest = pixel
329          elif hsv_state == 1:
330              s.append(pixel)
331          elif hsv_state == 2:
332              v.append(pixel)
333  
334          if hsv_state != 2:
335              hsv_state += 1
336          else:
337              hsv_state = 0
338  
339      offset_of_median = 50
340  
341      h_median = np.median(h)
342      s_median = np.median(s)
343      v_median = np.median(v)
344  
345      h_highest = int(h_median + offset_of_median)
346      h_lowest = int(h_median - offset_of_median)
347  
348      s_highest = int(s_median + offset_of_median)
349      s_lowest = int(s_median - offset_of_median)
350  
351      v_highest = int(v_median + offset_of_median)
352      v_lowest = int(v_median - offset_of_median)
353      print(f"h_median: {h_median}, s_median: {s_median}, v_median: {v_median}")
354      return h_highest, h_lowest, s_highest, s_lowest, v_highest, v_lowest

358      try:
359          switch = cv2.getTrackbarPos('switch', 'window')
360      except:
361          switch = 0
362      if switch == 1:
363          lowerr = cv2.getTrackbarPos('lower', 'window')
364          higherr = cv2.getTrackbarPos('higher', 'window')
365          hsv_ = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
366          lower_blue_ = np.array([lowerr, 0, 0])
367          upper_blue_ = np.array([higherr, 255, 255])
368  
369          # Here we are defining range of bluecolor in HSV
370          # This creates a mask of blue coloured
371          # objects found in the frame.
372          mask_ = cv2.inRange(hsv_, lower_blue_, upper_blue_)
373          mask_ = cv2.medianBlur(mask_, 5)
374  
375          # The bitwise and of the frame and mask is done so
376          # that only the blue coloured objects are highlighted
377          # and stored in res
378          res_ = cv2.bitwise_and(frame, frame, mask=mask_)
379  
380          res_ = frame - res_
381  
382          edges = canny(res_)
383          combined_ = cv2.hconcat([res_, frame, edges])

391          higherr = cv2.getTrackbarPos('higher', 'window')
392          hsv_ = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
